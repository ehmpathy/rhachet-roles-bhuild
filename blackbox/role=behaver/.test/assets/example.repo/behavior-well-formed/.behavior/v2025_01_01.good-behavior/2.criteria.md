# usecase.1 = generate greeting

given('a valid user name')
  when('greeting is requested with formal format')
    then('returns "Good day, {name}"')
      sothat('formal contexts are supported')

given('a valid user name')
  when('greeting is requested with casual format')
    then('returns "Hey {name}!"')
      sothat('informal contexts are supported')

# usecase.2 = handle edge cases

given('an empty name')
  when('greeting is requested')
    then('returns "Hello, friend"')
      sothat('missing names are handled gracefully')

given('a name with special characters')
  when('greeting is requested')
    then('sanitizes and uses the name')
      sothat('injection attacks are prevented')
