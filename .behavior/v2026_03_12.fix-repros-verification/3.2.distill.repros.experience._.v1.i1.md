## reproduction feasibility

### test utilities available

- `fs.readFileSync` to read emitted template content
- `expect(content).toContain(...)` for content assertions
- temp directory for isolated test environment
- BDD structure via `given/when/then` from `test-fns`

### setup required

- create temp directory
- call `initBehaviorDir` with temp path
- read emitted files
- assert on content

### concrete test sketch

```ts
given('[case5] repros stone and guard', () => {
  when('[t0] initBehaviorDir is called', () => {
    then('creates 3.2.distill.repros.experience._.v1.guard with 3 self-reviews', () => {
      initBehaviorDir({ behaviorDir, behaviorDirRel });

      const content = fs.readFileSync(
        path.join(behaviorDir, '3.2.distill.repros.experience._.v1.guard'),
        'utf-8',
      );

      expect(content).toContain('has-critical-paths-identified');
      expect(content).toContain('has-ergonomics-reviewed');
      expect(content).toContain('has-play-test-convention');
    });

    then('repros stone includes journey sketch guidance', () => {
      initBehaviorDir({ behaviorDir, behaviorDirRel });

      const content = fs.readFileSync(
        path.join(behaviorDir, '3.2.distill.repros.experience._.v1.stone'),
        'utf-8',
      );

      expect(content).toContain('given(');
      expect(content).toContain('when(');
      expect(content).toContain('then(');
      expect(content).toContain('snapshot');
      expect(content).toContain('critical path');
    });
  });
});

given('[case6] verification stone and guard enhancements', () => {
  when('[t0] initBehaviorDir is called', () => {
    then('verification stone references repros artifact', () => {
      initBehaviorDir({ behaviorDir, behaviorDirRel });

      const content = fs.readFileSync(
        path.join(behaviorDir, '5.3.verification.v1.stone'),
        'utf-8',
      );

      expect(content).toContain('3.2.distill.repros');
    });

    then('verification guard has 9 total self-reviews', () => {
      initBehaviorDir({ behaviorDir, behaviorDirRel });

      const content = fs.readFileSync(
        path.join(behaviorDir, '5.3.verification.v1.guard'),
        'utf-8',
      );

      // original 4
      expect(content).toContain('has-behavior-coverage');
      expect(content).toContain('has-zero-test-skips');
      expect(content).toContain('has-all-tests-passed');
      expect(content).toContain('has-preserved-test-intentions');

      // new 5
      expect(content).toContain('has-journey-tests-from-repros');
      expect(content).toContain('has-snapshot-coverage');
      expect(content).toContain('has-critical-paths-frictionless');
      expect(content).toContain('has-ergonomics-validated');
      expect(content).toContain('has-play-test-convention');
    });
  });
});
```
