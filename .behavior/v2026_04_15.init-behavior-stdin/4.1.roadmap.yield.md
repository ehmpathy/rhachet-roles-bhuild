### acceptance criteria

```
given(getWishContent transformer)
  when(wish === '@stdin')
    then(reads from fd 0)
  when(wish !== '@stdin')
    then(returns wish as-is)

given(findsertWishFromInput transformer)
  when(wishInput is empty)
    then(exit 2 with "error: --wish requires content")
  when(wishCurrent === wishExpected)
    then(no-op, idempotent)
  when(wishCurrent is template)
    then(writes wishExpected)
  when(wishCurrent differs)
    then(exit 2 with "error: wish file has been modified")
```
