# blueprint: --wish @stdin|words

## summary

add `--wish` flag to `init.behavior` skill that accepts:
- inline content: `--wish "my wish content"`
- stdin content: `--wish @stdin`

wish content populates `0.wish.md` before editor opens (if `--open` provided).

### getWishContent transformer

```ts
// src/domain.operations/behavior/init/getWishContent.ts
import { readFileSync } from 'fs';

/**
 * .what = extracts wish content from @stdin or returns inline value
 * .why = handles @stdin pattern for piped input
 */
export const getWishContent = (input: { wish: string }): string => {
  if (input.wish === '@stdin') {
    return readFileSync(0, 'utf-8').trim();
  }
  return input.wish;
};
```

### findsertWishFromInput transformer

```ts
// src/domain.operations/behavior/init/findsertWishFromInput.ts
import { readFileSync, writeFileSync } from 'fs';
import { relative } from 'path';

/**
 * .what = validates wish input and findserts into wish file
 * .why = findsert semantics: same = no-op, template = populate, different = error
 */
export const findsertWishFromInput = (input: {
  wishInput: string;
  wishPath: string;
}): void => {
  // validate non-empty
  if (!input.wishInput.trim()) {
    console.error('error: --wish requires content');
    process.exit(2);  // constraint error: user must fix
  }

  // findsert semantics
  const wishExpected = `wish =\n\n${input.wishInput}\n`;
  const wishCurrent = readFileSync(input.wishPath, 'utf-8');
  const wishPathRel = relative(process.cwd(), input.wishPath);

  if (wishCurrent === wishExpected) {
    // already populated with same content — no-op (idempotent)
    return;
  }

  if (wishCurrent.trim() === 'wish =') {
    // template — populate
    writeFileSync(input.wishPath, wishExpected);
    return;
  }

  // different content — error
  console.error('error: wish file has been modified');
  console.error('');
  console.error('to overwrite, delete the wish file first:');
  console.error(`  rm ${wishPathRel}`);
  process.exit(2);  // constraint error: user must fix
};
```

```ts
// blackbox/role=behaver/.test/skill.init.behavior.utils.ts
export const runInitBehaviorSkillDirect = (input: {
  args: string;
  repoDir: string;
  stdin?: string;  // [+] optional stdin param
}): { stdout: string; stderr: string; exitCode: number } => {
  try {
    const stdout = execSync(`bash "${SCRIPT_PATH}" ${input.args}`, {
      cwd: input.repoDir,
      encoding: 'utf-8',
      ...(input.stdin && { input: input.stdin }),  // [+] pass stdin if provided
      env: { ...process.env, PATH: process.env.PATH },
    });
    return { stdout: stdout.trim(), stderr: '', exitCode: 0 };
  } catch (error: unknown) {
    // ... extant error handler
  }
};
```
