## pattern 1: genConsumerRepo

**[REUSE]**

### citation [1]: `blackbox/.test/infra/genConsumerRepo.ts`

```ts
export const genConsumerRepo = (): { repoDir: string; cleanup: () => void } => {
  // create temp directory
  const repoDir = mkdtempSync(join(tmpdir(), 'test-consumer-'));

  // init git repo
  execSync('git init', { cwd: repoDir, stdio: 'pipe' });
  execSync('git config user.email "test@test.com"', { cwd: repoDir, stdio: 'pipe' });
  execSync('git config user.name "Test"', { cwd: repoDir, stdio: 'pipe' });

  // symlink node_modules from main repo
  const mainRepoNodeModules = join(__dirname, '..', '..', '..', 'node_modules');
  symlinkSync(mainRepoNodeModules, join(repoDir, 'node_modules'));

  // create package.json
  writeFileSync(join(repoDir, 'package.json'), JSON.stringify({ name: 'test-consumer', version: '1.0.0' }));

  // initial commit
  execSync('git add .', { cwd: repoDir, stdio: 'pipe' });
  execSync('git commit -m "init"', { cwd: repoDir, stdio: 'pipe' });

  return {
    repoDir,
    cleanup: () => rmSync(repoDir, { recursive: true, force: true }),
  };
};
```

### relevance

use for all --wish acceptance tests. provides:
- isolated temp git repo
- symlinked node_modules (access to rhachet)
- cleanup function for afterAll

## pattern 2: runInitBehaviorSkillDirect

**[REUSE]**

### citation [2]: `blackbox/role=behaver/.test/skill.init.behavior.utils.ts` lines 12-32

```ts
export const runInitBehaviorSkillDirect = (input: {
  args: string;
  repoDir: string;
}): { stdout: string; exitCode: number } => {
  try {
    const stdout = execSync(`bash "${SCRIPT_PATH}" ${input.args}`, {
      cwd: input.repoDir,
      utf8: true,
      env: { ...process.env, PATH: process.env.PATH },
    });
    return { stdout: stdout.trim(), exitCode: 0 };
  } catch (error: unknown) {
    if (error && typeof error === 'object' && 'status' in error) {
      const execError = error as { status: number; stdout?: Buffer | string };
      return {
        stdout: execError.stdout?.toString().trim() ?? '',
        exitCode: execError.status,
      };
    }
    throw error;
  }
};
```

### relevance

direct skill invocation for acceptance tests. returns:
- stdout for output assertions
- exitCode for error case tests

## pattern 4: asSnapshotStable

**[REUSE]**

### citation [4]: `blackbox/role=behaver/.test/skill.init.behavior.utils.ts` lines 34-47

```ts
export const asSnapshotStable = (input: { output: string; repoDir: string }): string => {
  const repoDirEscaped = input.repoDir.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
  return input.output
    .replace(new RegExp(repoDirEscaped, 'g'), '/tmp/test-consumer-XXXX')
    .replace(/v\d{4}_\d{2}_\d{2}\.[a-z0-9-]+/g, 'vYYYY_MM_DD.branch-name')
    .replace(/\d{4}_\d{2}_\d{2}/g, 'YYYY_MM_DD');
};
```

### relevance

sanitize output for snapshot stability:
- replaces temp dir paths
- replaces date stamps
- enables deterministic snapshots

## pattern 5: BDD test structure

**[REUSE]**

### citation [5]: `blackbox/role=behaver/skill.init.behavior.flags.acceptance.test.ts` lines 1-50

```ts
import { given, then, useBeforeAll, when } from 'test-fns';
import { genConsumerRepo } from '../.test/infra/genConsumerRepo';
import { asSnapshotStable, runInitBehaviorSkillDirect } from './.test/skill.init.behavior.utils';

describe('skill.init.behavior flags', () => {
  const consumer = useBeforeAll(() => genConsumerRepo());
  afterAll(() => consumer.cleanup());

  given('[case1] --size nano', () => {
    when('[t0] init.behavior is invoked', () => {
      const result = useBeforeAll(() =>
        runInitBehaviorSkillDirect({
          args: '--name test-nano --size nano',
          repoDir: consumer.repoDir,
        }),
      );

      then('exit code is 0', () => {
        expect(result.exitCode).toBe(0);
      });

      then('output matches snapshot', () => {
        expect(asSnapshotStable({ output: result.stdout, repoDir: consumer.repoDir })).toMatchSnapshot();
      });
    });
  });
});
```

### relevance

follow exact structure for --wish tests:
- single describe block
- useBeforeAll for consumer repo
- afterAll for cleanup
- given/when/then blocks
- exitCode and snapshot assertions
