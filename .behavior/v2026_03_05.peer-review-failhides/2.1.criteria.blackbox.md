# criteria.blackbox: peer-review failhides

## usecase.1 = blueprint phase peer review

given(behaver completes blueprint with error-handle prose)
  when(behaver runs `rhx route.stone.set --stone 3.3.blueprint.v1 --as passed`)
    then(peer review scans blueprint files for failhide patterns in changed files)
      sothat(error-swallow designs are caught before code is written)
    then(review output is written to `.behavior/$route/.reviews/$stone.peer-review.failhides.md`)
      sothat(behaver can see exactly what was flagged)

given(blueprint describes try/catch that swallows errors)
  when(peer review runs)
    then(review detects the failhide pattern in prose)
    then(stone is blocked until behaver fixes the design)
      sothat(failhide designs don't proceed to execution)

given(blueprint has no failhide patterns)
  when(peer review runs)
    then(review passes with clean status)
    then(stone proceeds to next guard check)

---

## usecase.2 = execution phase peer review

given(behaver completes execution with code changes in src/)
  when(behaver runs `rhx route.stone.set --stone 5.1.execution.* --as passed`)
    then(peer review scans src/ files for failhide patterns in changed files)
      sothat(error-swallow code is caught before merge)
    then(review output is written to `.behavior/$route/.reviews/$stone.peer-review.failhides.md`)

given(code has try/catch that swallows errors without rethrow)
  when(peer review runs)
    then(review detects the failhide pattern)
    then(review report shows file:line and code snippet)
    then(stone is blocked until behaver fixes the code)
      sothat(failhides don't reach production)

given(code has try/catch that properly handles errors)
  when(peer review runs)
    then(review passes — legitimate error handle is not flagged)
      sothat(false positives are minimized)

---

## usecase.3 = review output format

given(peer review detects failhides)
  when(review completes)
    then(report shows blocker count)
    then(report shows each violation with file path and line number)
    then(report shows code snippet of the violation)
      sothat(behaver can quickly locate and fix the issue)

given(peer review detects no failhides)
  when(review completes)
    then(report shows "all clear" status)
    then(report shows 0 blockers, 0 nitpicks)

---

## usecase.4 = scope behavior

given(behaver changes files in src/ and .behavior/)
  when(blueprint peer review runs with `--paths-with ".behavior/$route/3.3.blueprint.*.md" --join intersect`)
    then(only blueprint files in the diff are reviewed)
    then(src/ changes are not reviewed by blueprint guard)
      sothat(each phase reviews only its relevant files)

given(behaver changes files in src/ and .behavior/)
  when(execution peer review runs with `--paths-with "src/**/*.ts" --join intersect`)
    then(only src/ files in the diff are reviewed)
    then(.behavior/ changes are not reviewed by execution guard)
