## reproduction feasibility

### test utilities available

- `fs.mkdtempSync` — temp directory creation
- `fs.readFileSync` — template content verification
- `fs.existsSync` — file presence check
- `given/when/then` from `test-fns` — BDD structure

### setup required

```ts
beforeEach(() => {
  tempDir = fs.mkdtempSync(path.join(os.tmpdir(), 'reflection-test-'));
  behaviorDir = path.join(tempDir, '.behavior', 'v2025_01_01.test-feature');
});

afterEach(() => {
  fs.rmSync(tempDir, { recursive: true, force: true });
});
```

### concrete test sketch

```ts
given('[case] reflection stone templates', () => {
  when('[t0] initBehaviorDir is called', () => {
    then('creates audience reflection stone', () => {
      const result = initBehaviorDir({
        behaviorDir,
        behaviorDirRel: '.behavior/v2025_01_01.test-feature',
      });

      expect(result.created).toContain('3.1.5.research.reflection.product.audience._.v1.stone');

      const content = fs.readFileSync(
        path.join(behaviorDir, '3.1.5.research.reflection.product.audience._.v1.stone'),
        'utf-8',
      );
      expect(content).toContain('who is the audience');
      expect(content).toContain('why do they care');
      expect(content).toContain('how much do they care');
    });

    then('creates rootcause reflection stone', () => {
      const result = initBehaviorDir({
        behaviorDir,
        behaviorDirRel: '.behavior/v2025_01_01.test-feature',
      });

      expect(result.created).toContain('3.1.5.research.reflection.product.rootcause._.v1.stone');

      const content = fs.readFileSync(
        path.join(behaviorDir, '3.1.5.research.reflection.product.rootcause._.v1.stone'),
        'utf-8',
      );
      expect(content).toContain('5 whys');
      expect(content).toContain('root cause');
      expect(content).toContain('symptom');
    });

    then('creates premortem reflection stone', () => {
      const result = initBehaviorDir({
        behaviorDir,
        behaviorDirRel: '.behavior/v2025_01_01.test-feature',
      });

      expect(result.created).toContain('3.1.5.research.reflection.product.premortem._.v1.stone');

      const content = fs.readFileSync(
        path.join(behaviorDir, '3.1.5.research.reflection.product.premortem._.v1.stone'),
        'utf-8',
      );
      expect(content).toContain('failed');
      expect(content).toContain('assumptions');
      expect(content).toContain('nightmare scenario');
    });
  });
});
```
