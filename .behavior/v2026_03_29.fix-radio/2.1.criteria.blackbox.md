# blackbox criteria: fix-radio keyrack integration

experience boundaries for the radio skill keyrack integration.

---

# usecase.1 = radio skill just works

given(developer has filled keyrack with ehmpath creds)
given(repo has been initialized with dispatcher role keys)
  when(developer runs `rhx radio.task.push --via gh.issues --title "..." --description "..."`)
    then(task is created in github issues)
      sothat(developer broadcasts work without manual auth setup)
    then(output shows task exid, status, repo, and channel)
      sothat(developer can track the created task)
    then(no --auth flag is required)
      sothat(developer experience is frictionless)

  when(developer runs `rhx radio.task.pull --via gh.issues --exid 42 --status CLAIMED`)
    then(task status is updated in github issues)
      sothat(developer claims work from distributed queue)
    then(output confirms the status transition)

# usecase.2 = machine-wide credential setup

given(developer has not filled keyrack yet)
  when(developer runs `rhx keyrack fill --owner ehmpath --env prep`)
    then(credentials are stored in machine-wide vault)
      sothat(setup is one-time per machine)
    then(credentials are available to all repos that init keys)
      sothat(no per-repo credential duplication)

given(developer has filled keyrack)
  when(developer runs `rhx keyrack fill --owner ehmpath --env prep` again)
    then(credentials are updated idempotently)
      sothat(refresh is safe to run multiple times)

# usecase.3 = per-repo key initialization

given(developer has filled keyrack machine-wide)
  when(developer runs `npx rhachet init --keys --roles dispatcher`)
    then(repo gains access to dispatcher role credentials)
      sothat(radio skills can authenticate)
    then(keyrack.yml is linked for the repo)
      sothat(credential requirements are declared)

given(repo has not initialized keys)
  when(developer runs radio skill)
    then(clear error indicates keys not initialized)
      sothat(developer knows to run init --keys)

# usecase.4 = test credential flow

given(test env has keyrack.yml with EHMPATH_BEAVER_GITHUB_TOKEN)
given(test runner unlocks keyrack)
  when(integration test runs radio skill)
    then(test uses ephemeral token from keyrack)
      sothat(tests use production-like credential path)
    then(no manual env var setup required in test)
      sothat(test setup is minimal)

# usecase.5 = ci/cd keyrack integration

given(ci/cd workflow has `npm run prepare:rhachet` step)
given(keyrack.source() runs before tests)
  when(radio skill runs in ci/cd)
    then(skill uses keyrack-sourced credentials)
      sothat(ci/cd uses same keyrack path as local dev)
    then(behavior is identical to local dev)
      sothat(ci/cd and local dev are consistent)

# usecase.6 = error experiences

given(keyrack is not unlocked)
  when(developer runs radio skill)
    then(clear error: "keyrack not unlocked for env=prep")
      sothat(developer knows to unlock keyrack)

given(EHMPATH_BEAVER_GITHUB_TOKEN not in vault)
  when(developer runs radio skill)
    then(clear error: "EHMPATH_BEAVER_GITHUB_TOKEN not found in vault")
      sothat(developer knows to fill keyrack)

given(keyrack token has expired)
  when(developer runs radio skill)
    then(keyrack auto-refreshes token transparently)
      sothat(developer never sees expiry errors)

# usecase.7 = ephemeral token security

given(developer runs radio skill)
  when(keyrack unlocks EHMPATH_BEAVER_GITHUB_TOKEN)
    then(returned token is short-lived (ghs_* format))
      sothat(security is improved over long-lived PATs)
    then(token expires in ~1 hour)
      sothat(leaked tokens have limited blast radius)
    then(github app json is never exposed to skill)
      sothat(master credentials stay protected)

# usecase.8 = github app provisioned

given(ehmpath-beaver github app does not exist)
  when(developer provisions via `@repo/provision/ehmpath-beaver`)
    then(github app is created with issues:write permission)
      sothat(radio skill can create/update issues)
    then(app is public)
      sothat(any repo can install it)
    then(credentials (appId, privateKey, installationId) are produced)
      sothat(keyrack can store and translate them)

# usecase.9 = ci/cd keyrack preparation

given(ci/cd workflow runs tests)
  when(integration or acceptance tests are about to run)
    then(`npm run prepare:rhachet` runs first)
      sothat(keyrack is prepared with credentials)
    then(tests can access EHMPATH_BEAVER_GITHUB_TOKEN)
      sothat(tests pass in ci/cd environment)
