## subcomponent contracts

given('ehmpath-beaver github app')
  then('is provisioned via declastruct-github pattern')
  then('is a public github app')
  then('has permissions: issues:write')
  then('has installation on target repos')
  then('produces: appId, privateKey, installationId')

given('dispatcher role keyrack.yml')
  then('declares EHMPATH_BEAVER_GITHUB_TOKEN')
  then('specifies env=prep')
  then('specifies mechanism=EPHEMERAL_VIA_GITHUB_APP')
  then('is registered on getDispatcherRole')

given('.agent/keyrack.yml for test env')
  then('declares EHMPATH_BEAVER_GITHUB_TOKEN')
  then('specifies env=test')
  then('specifies mechanism=EPHEMERAL_VIA_GITHUB_APP')

given('radio skill auth')
  then('tries keyrack first: `keyrack unlock --owner ehmpath --env prep --key EHMPATH_BEAVER_GITHUB_TOKEN`')
  then('falls back to explicit --auth if keyrack unavailable')
  then('returns ephemeral ghs_* token when keyrack succeeds')
  then('returns explicit --auth token when keyrack unavailable')
  then('throws clear error when neither available')
